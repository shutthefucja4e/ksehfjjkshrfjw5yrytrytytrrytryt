Tab:CreateButton({
   Name = "Auto Trading Plaza",
   Callback = function()
      local plazaZone = game.Workspace.Zones.PlazaZone
      if plazaZone then
         game.Players.LocalPlayer.Character:SetPrimaryPartCFrame(plazaZone.CFrame)
      else
         Rayfield:Notify({
            Title = "Ошибка",
            Content = "PlazaZone не найдена",
            Duration = 3,
         })
      end
   end,
})
